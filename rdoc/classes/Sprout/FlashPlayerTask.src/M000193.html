    def FlashPlayerTask.home
      if(@@home)
        return @@home
      end

      FlashPlayerTask.home_paths.each do |path|
        if(File.exists?(path))
          return @@home = path
        end
      end

      if(@@home.nil?)
        raise FlashPlayerError.new('FlashPlayer unable to find home folder for your platform')
      end
      return @@home
    end
