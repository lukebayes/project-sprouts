def flush
  @bars.keys.each do |title|
    print(title)
  end
  
  @bars.values.each do |bar|
    if(bar.finished?)
      print(bar.title)
      outio.print "\n"
      @outs.delete(bar.title)
      @bars.delete(bar.title)
    end
  end
end
